bool is_prime(std::size_t n)
{
	if (n <= 1) return false;

	if (n <= 3) return true;

	if (n % 2 == 0 || n % 3 == 0) return false;

	for (std::size_t m = 5, stop = std::sqrt(n); m <= stop; m += 6)
	{
		if (n % m == 0) return false;

		if (n % (m + 2) == 0) return false;
	}

	return true;
}
